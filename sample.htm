Customization Example

<!-- Add jQuery here -->

<script src="jquery.slidereference.js"></script>

<script>

// >> After Page Load
$(function(){

// >> Slidereference event handler
$(".slidereference").bind("click", function(){

// Optional Toggle Btn Text
if ($(this).text() == 'Deactivate'){
$(this).text('Activate');
} else {
$(this).text('Deactivate');
// Clear previews
$('#slidereference_div').remove();
}

// Create new slide
$('<div/>')
.prop('id','slidereference_div')
.appendTo('body');

// Run plugin
$('#slidereference_div').slidereference({
'url' : 'http://www.google.com/',
loading_text : "Please wait...",
ownsite_width : "60%",
reference_width : "40%",
'loadingcss' : {
"background-color" : "red"
}
});

});
// << Slidereference event handler

});
// << After Page Load

</script>
